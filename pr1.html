Applet program to draw a face
    import java.applet.Applet;
    import java.awt.*;
    
    /*
    < applet code = "Face.class" width = "600" height = "600">
    */ 
    
    public class Face extends Applet {
        public void paint(Graphics g) {
            g.drawOval(50, 50, 150, 150);
    
            // Eyes
            g.setColor(Color.black);
            g.fillOval(80, 100, 10, 15);
            g.fillOval(160, 100, 10, 15);
    
            // Nose
            g.drawOval(120, 135, 8, 13);
    
            // Smile
            g.drawArc(101, 150, 50, 20, 180, 180);
        }
    }
